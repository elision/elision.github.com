    Parse an atom with the exception of a map pair or general application. Most atoms are matched at this point.

    FirstAtom ::= "(" Atom ")"
            | "^TYPE"
            | ParsedVariable
            | "\" ParsedVariable "." FirstAtom
            | "{:" Atom Atom ":}"

            | "%" ParsedAlgProp ("(" ParsedRawAtomSeq ")")?
            | ESymbol ( "(" ParsedRawAtomSeq ")"

                      )?
